15 * |
16 * ------------------------------------
17 * |
18 * v
19 * type_list {
20 * uint size, # Size of the list in entries
21 * list type_item[size] # ---------|
22 * } |
23 * |
24 * -------------------------------------
25 * |
26 * v
27 * type_item {
28 * ushort type_idx # type_idx of each member
29 * }
30 *
31 * ProtoID[] protos
32 */
