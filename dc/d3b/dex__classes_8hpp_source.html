1/***
2 * @file dex_classes.hpp
3 * @author @Farenain
4 *
5 * @brief Android classes of the java code, composed
6 * by different fields:
7 *
8 * ClassDefsStruct{
9 * uint class_idx, // index into type_ids list, must be a class type not array or primitive type.
10 * uint access_flags, // access flags for class (public, final, etc)
11 * uint superclass_idx, // index into type_ids list for super class, or constant value NO_INDEX if has no superclass.
12 * uint interfaces_off, // offset from start of the file to list of interfaces, or 0 if there are none. Offset should be in data section.
13 * uint source_file_idx, // index into string_ids for name of file containing original source, or value NO_INDEX if no information.
14 * uint annotations_off, // offset from start of file to annotations structure for this class, or 0 if no annotations.
15 * uint class_data_off, // offset from start of the file to the associated class data for this item, or 0 if no class data.
16 * uint static_values_off // offset from the start of the file to the list of initial values for static fields, or 0 if there are none.
17 * }
18 *
19 * interfaces_off points to:
20 * list of Class type
21 *
22 * class_data_off points to:
23 *
24 * ClassDataItem {
25 * uleb128 static_fields_size, // number of static fields defined in item
26 * uleb128 instance_fields_size, // number of instance fields defined in item
27 * uleb128 direct_methods_size, // number of direct methods defined in item
28 * uleb128 virtual_methods_size, // number of virtual methods defined in item
29 * encoded_field[static_fields_size] static_fields, // defined static fields, fields must be sorted by field_idx
30 * encoded_field[instance_fields_size] instance fields, // defined instance fields, fields must be sorted by field_idx
31 * encoded_method[direct_methods_size] direct_methods, // defined (static, private or constructor) methods. Must be sorted by method_idx in increasing order.
32 * encoded_method[virtual_methods_size] virtual_methods // defined virtual (static, private or constructor) methods. Must be sorted by method_idx in increasing order.
33 * }
34 */
